const stations = [
  {
    freq: 91.5,
    name: "DZIO Hit Radio",
    rds: "HIT RADIO • DZIO-FM •",
    songs: [
      {title:"Pop Hit 1", src:"audio1.mp3"},
      {title:"Pop Hit 2", src:"audio2.mp3"}
    ]
  },
  {
    freq: 95.5,
    name: "Chill Radio",
    rds: "CHILL RADIO • RELAX MUSIC •",
    songs: [
      {title:"Chill 1", src:"audio3.mp3"},
      {title:"Chill 2", src:"audio4.mp3"}
    ]
  }
];

const player = document.getElementById("player");
const stationName = document.getElementById("stationName");
const songName = document.getElementById("songName");
const tuner = document.getElementById("tuner");
const freqDisplay = document.getElementById("freq");
const staticFx = document.getElementById("static");
const rdsText = document.getElementById("rdsText");

function startRadio(station){
  stationName.textContent = station.name;
  rdsText.textContent = station.rds;

  function playRandom(){
    const track = station.songs[Math.floor(Math.random()*station.songs.length)];
    player.src = track.src;
    songName.textContent = "Now Playing: " + track.title;
    rdsText.textContent = station.rds + " " + track.title;
    player.play().catch(()=>{});
    player.onended = playRandom;
  }

  playRandom();
}

function tuneToFrequency(value){
  freqDisplay.textContent = value.toFixed(1) + " FM";
  staticFx.style.opacity = 0.4;
  player.pause();
  songName.textContent = "Searching signal...";
  rdsText.textContent = "...";

  let found = stations.find(s => Math.abs(s.freq - value) < 0.3);

  setTimeout(()=>{
    staticFx.style.opacity = 0;

    if(found){
      startRadio(found);
    } else {
      stationName.textContent = "No Signal";
      songName.textContent = "Static noise...";
      rdsText.textContent = "NO SIGNAL";
    }

  }, 1000);
}

tuner.addEventListener("input", ()=>{
  tuneToFrequency(parseFloat(tuner.value));
});

tuner.value = 87.5;
tuneToFrequency(87.5);
